\frametitle{Project Goals}
%--------------------------------------------------

\enhance{1}\item data structure scalability (zones, grids, levels)
\end{itemize}
\enhance{2}\item{Easy to use}
\begin{itemize}

\enhance{2}\item enhanced parameter file grammar and syntax

\enhance{3}\item loosely-coupled components
\end{itemize}
\end{itemize}
